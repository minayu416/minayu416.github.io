確認連結資料庫 (Complete)

#!/usr/bin/env python

# -*- coding: utf-8 -*-

import pymysql

# 创建连接

conn = pymysql.connect(host='hostip', port=xxxx, user='user', passwd='pw', db='db')

# 开启自动提交SQL，如果这里不设置，以后的命令需要执行conn.commit()来提交执行，否则都在内存中

conn.autocommit(True)

# 创建游标

cur = conn.cursor()

# 执行普通SQL，并返回受影响行数

effect_row = cur.execute("show tables")

print(effect_row) 

# 提交

conn.commit()

# 关闭游标

cur.close()

# 关闭连接

conn.close()
