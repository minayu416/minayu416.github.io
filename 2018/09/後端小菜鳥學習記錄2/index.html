def insert_record():

    # 自動傳送SQL

    # 上傳

    Name = db.Column(db.String(64))

    def__init__(self, name, time, value, boolean): 
        self.Name = name 
        self.Time = time 
        self.Value = value 
        self.Boolean = boolean
